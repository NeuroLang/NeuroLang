.. GENERATED FROM PYTHON SOURCE LINES 81-96

    class Datalog(
        region_solver.RegionSolver,
        TranslateToLogic,
        DatalogProgram,
        ew.ExpressionBasicEvaluator
    ):
        def function_lh(self, x: str) -> bool:
            return x.startswith('L ')

        def function_rh(self, x: str) -> bool:
            return x.startswith('R ')
